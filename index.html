import tkinter as tk from tkinter import ttk, messagebox, filedialog import threading import yt_dlp import os class SnaptubePC: def __init__(self, root): self.root = root self.root.title("منصة التحميل الذكية للكمبيوتر (SnapTube PC)") self.root.geometry("600x350") self.root.configure(bg="#0f172a") # خلفية داكنة احترافية # ضبط اتجاه الواجهة والخطوط self.font_main = ("Cairo", 12) self.font_title = ("Cairo", 16, "bold") # العنوان العلوي title_lbl = tk.Label(root, text="محرك تحميل يوتيوب الاحترافي", font=self.font_title, fg="#f8fafc", bg="#0f172a") title_lbl.pack(pady=20) # حقل إدخال الرابط input_frame = tk.Frame(root, bg="#0f172a") input_frame.pack(pady=10, fill="x", px=40) lbl_url = tk.Label(input_frame, text="أدخل رابط الفيديو أو قائمة التشغيل (Playlist):", font=self.font_main, fg="#94a3b8", bg="#0f172a") lbl_url.pack(anchor="e", pady=5) self.url_entry = tk.Entry(input_frame, font=("Arial", 11), bg="#1e293b", fg="#f8fafc", insertbackground="white", bd=0, relief="flat") self.url_entry.pack(fill="x", ipady=12) # شريط نسبة تقدم التحميل self.progress_bar = ttk.Progressbar(root, orient="horizontal", length=400, mode="determinate") self.progress_bar.pack(pady=20) self.status_lbl = tk.Label(root, text="جاهز لبدء التحميل...", font=("Cairo", 10), fg="#94a3b8", bg="#0f172a") self.status_lbl.pack() # زر التحميل المباشر self.download_btn = tk.Button(root, text="ابدأ التحميل وحدد مسار الحفظ", font=self.font_main, bg="#3b82f6", fg="white", bd=0, relief="flat", activebackground="#2563eb", activeforeground="white", command=self.start_download_thread) self.download_btn.pack(pady=15, ipady=8, ipadx=20) def progress_hook(self, d): """دالة تحديث شريط التقدم ونسبة التحميل في الخلفية""" if d['status'] == 'downloading': # استخراج النسبة المئوية للتحميل total_bytes = d.get('total_bytes') or d.get('total_bytes_estimate') or 1 downloaded_bytes = d.get('downloaded_bytes', 0) percentage = (downloaded_bytes / total_bytes) * 100 # تحديث الواجهة بأمان self.progress_bar['value'] = percentage self.status_lbl.config(text=f"جاري التحميل في الخلفية: {percentage:.1f}% | السرعة: {d.get('_speed_str', '0B/s')}") self.root.update_idletasks() elif d['status'] == 'finished': self.status_lbl.config(text="✨ اكتمل تحميل الملف ويتم الآن الحفظ النهائي بجهازك...") def download_process(self, url, save_path): """المحرك الخلفي المسؤول عن الاتصال المباشر والتحميل الفعلي""" ydl_opts = { # تحميل أفضل جودة فيديو مدمجة مع الصوت مباشرة لتفادي التعقيدات 'format': 'bestvideo[ext=mp4]+bestaudio[ext=m4a]/best[ext=mp4]/best', 'outtmpl': os.path.join(save_path, '%(title)s.%(ext)s'), 'progress_hooks': [self.progress_hook], 'noplaylist': False, # السماح بتحميل القوائم الكاملة تلقائياً } try: with yt_dlp.YoutubeDL(ydl_opts) as ydl: ydl.download([url]) messagebox.showinfo("نجاح العملية", "✨ تم تحميل وفصل ودمج الملفات بنجاح وحفظها في المجلد المحدد!") self.status_lbl.config(text="جاهز لبدء التحميل...") self.progress_bar['value'] = 0 except Exception as e: messagebox.showerror("خطأ في المحرك", f"تعذر التحميل، تأكد من اتصالك بالإنترنت ومن صحة الرابط.\nالخطأ: {str(e)}") self.status_lbl.config(text="حدث خطأ أثناء المعالجة.") finally: self.download_btn.config(state="normal") def start_download_thread(self): url = self.url_entry.get().strip() if not url: messagebox.showwarning("تنبيه", "من فضلك ضع رابط يوتيوب أولاً!") return # تفتح نافذة نظام الويندوز الرسمية ليختار المستخدم مكان حفظ الملف بنفسه وبكل حرية save_directory = filedialog.askdirectory(title="اختر المجلد الذي تريد حفظ الفيديوهات بداخله") if not save_directory: return # إذا أغلق المستخدم النافذة بدون اختيار مجلد self.download_btn.config(state="disabled") self.status_lbl.config(text="جاري الاتصال المباشر بخوادم يوتيوب وفك التشفير...") # تشغيل دالة التحميل في مسار خلفي منفصل (Thread) لمنع تجمد الواجهة أثناء تنزيل الأحجام الكبيرة threading.Thread(target=self.download_process, args=(url, save_directory), daemon=True).start() if __name__ == "__main__": root = tk.Tk() # لتعديل مظهر شريط التقدم ليكون فلات وحديث style = ttk.Style() style.theme_use('default') style.configure("Horizontal.TProgressbar", thickness=8, background="#10b981", trowcolor="#1e293b") app = SnaptubePC(root) root.mainloop()
